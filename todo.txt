        WhenDamaged(Entity source, Entity receiver, Damage damage, Effect effect)
        Effect -> TimedEffect(ShieldEffect(DamageType type (damage.getType()), Amount amount), ticks=5, receiver)
        receiver.ifPresent(e -> effect.apply(e))

        -> ThornyElementalShield: Combination of thorns and elemental shield
        WhenDamaged(List<Effect> sourceEffects, List<Effect> receiverEffects) {
            trigger(source, receiver, stage, damage) {
                sourceEffects.forEach(effect -> effect.apply(source);
                receiverEffects.forEach(effect -> effect.apply(receiver);
            }
        }
